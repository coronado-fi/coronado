102    def _action(klass, transactionID: str, offerID: str, notes:str = None, action:str = 'approve') -> object:
103        if action not in ( 'approve', 'deny'):
104            raise CallError('allowed actions:  approve, deny')
105
106        spec = {
107            'transaction_id': transactionID,
108            'offer_id': offerID,
109        }
110        if notes:
111            spec['notes'] = notes
112
113        if None == notes and 'deny' == action:
114            raise CallError('notes attribute missing or set to None')
115
116        if '' == notes and 'deny' == action:
117            raise CallError('notes attribute must have some text; empty strings disallowed')
118
119        endpoint = '/'.join([ klass._serviceURL, 'partner/rewards.%s' % action ])
120        response = requests.request('POST', endpoint, headers = klass.headers, json = spec)
121
122        if response.status_code == 200:
123            result = _assembleDetailsFrom(response.content)
124        elif response.status_code == 404:
125            result = False
126        else:
127            raise errorFor(response.status_code, response.text)
128
129        return result
