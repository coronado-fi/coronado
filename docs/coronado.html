 80    def __init__(self, obj = None):
 81        """
 82        Create a new instance of a triple object.  `obj` must correspond to a
 83        valid, existing object ID if it's not a collection or JSON.  The
 84        constructor only returns a valid object if a subclass is instantiated;
 85        TripleObject is an abstract class, and passing it an object ID will
 86        raise an error.
 87
 88        Arguments
 89        ---------
 90            obj
 91        An object used for building a valid triple object.  The object can
 92        be one of:
 93
 94        - A dictionary - a dictionary with instantiation values as described
 95          in the API documentation
 96        - A JSON string
 97        - A triple objectID
 98
 99        Raises
100        ------
101            CoronadoError
102        A CoronadoError dependent on the specific error condition.  The full list of
103        possible errors, causes, and semantics is available in the 
104        **`coronado.exceptions`** module.
105        """
106        if isinstance(obj, str):
107            if '{' in obj:
108                d = json.loads(obj)
109            else:
110                try:
111                    d = self.__class__.byID(obj).__dict__
112                except Exception:  # Python, triple, or Coronado exceptions can happen here; unsure of how it'll blow up
113                    raise CallError('obj initializer is set to None or an invalid value type; obj = %s ' % obj)
114        elif isinstance(obj, dict):
115            d = deepcopy(obj)
116        elif isinstance(obj, TripleObject):
117            d = deepcopy(obj.__dict__)
118        else:
119            raise CallError('Invalid constructor obj type; it must be a string, a JSON payload, a dictionary, or a TripleObject')
120
121        d = tripleKeysToCamelCase(d)
122
123        for key, value in d.items():
124            if isinstance(value, (list, tuple)):
125                setattr(self, key, [TripleObject(x) if isinstance(x, dict) else x for x in value])
126            else:
127                setattr(self, key, TripleObject(value) if isinstance(value, dict) else value)
128
129        self.assertAll()

132    def assertAll(self) -> bool:
133        """
134        Asserts that all the attributes listed in the `requiredAttributes` list
135        of attribute names are presein the final object.  Coronado/triple
136        objects are built from JSON inputs which may or may not include all
137        required attributes.  This method ensures they do.
138
139        Returns
140        -------
141            True if all required attributes are present during initialization
142
143        Raises
144        ------
145            CoronadoError
146        A CoronadoError dependent on the specific error condition.  The full list of
147        possible errors, causes, and semantics is available in the 
148        **`coronado.exceptions`** module.
149        """
150        if self.__class__.requiredAttributes:
151            attributes = self.__dict__.keys()
152            if not all(attribute in attributes for attribute in self.__class__.requiredAttributes):
153                missing = set(self.__class__.requiredAttributes)-set(attributes)
154                raise CallError("attribute%s %s missing during instantiation" % ('' if len(missing) == 1 else 's', missing))

446        """
447        Return a list of tripleObjects.  The list is a sequential query from the
